PYTHON EXAMPLES

Load and analyze the delay data:

    import pandas as pd
    
    # Load 5.9 million train records
    df = pd.read_parquet('data/synthetic/train_delays_synthetic_2025_2026.parquet')
    
    # Basic statistics
    print(f"Total records: {len(df):,}")
    print(f"Delay rate: {df['has_delay'].mean()*100:.2f}%")
    print(f"Avg delay: {df[df['has_delay']==1]['delay_minutes'].mean():.1f} min")
    
    # Analyze by train type
    print(df['train_type'].value_counts(normalize=True))
    
    # Analyze by delay cause
    print(df[df['has_delay']==1]['delay_cause'].value_counts())
    
    # Time series analysis
    df['date'] = pd.to_datetime(df['date'])
    daily = df.groupby('date').agg({'has_delay': 'sum', 'delay_minutes': 'mean'})
